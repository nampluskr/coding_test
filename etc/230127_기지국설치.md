#include <iostream>
#include <map>
#include <algorithm>
#include <vector>
 
using namespace std;
 
struct Pos
{
	int pos, mid;
};
 
map<int, int> loc;
vector<Pos> pos;
 
void init(int N, int mId[], int mLocation[]) {
 
	loc.clear();
	pos.clear();
 
	for (int i = 0; i < N; i++) {
		loc[mId[i]] = i;
		pos.push_back({ mLocation[i], mId[i] });
	}
}
 
int add(int mId, int mLocation) {
 
	auto ret = loc.find(mId);
 
	if (ret != loc.end()) {
		pos[ret->second] = { mLocation, mId };
	} else {
		loc[mId] = pos.size();
		pos.push_back({ mLocation , mId });
	}
 
	return pos.size();
}
 
 
int remove(int mStart, int mEnd) {
 
	int id = 0;
	vector<Pos> newPos;
	map<int, int> newLoc;
 
	for (auto & p : pos)
	{
		if (mStart <= p.pos and p.pos <= mEnd)
			continue;
 
		newLoc[p.mid] = id++;
		newPos.push_back(p);
	}
 
	pos = newPos;
	loc = newLoc;
 
	return pos.size();
}
 
int install(int M) {
 
	vector<int> p(pos.size());
	for (int i = 0; i < pos.size(); i++)
		p[i] = pos[i].pos;
 
	sort(p.begin(), p.end(), [](const int& A, const int& B) ->bool { return A < B; });
 
	int e = p.back() - p.front(), s = 2, m;
 
	int ret = 0;
	while (s <= e)
	{
		m = (s + e) / 2;
 
		int n = 1;
		int total = 1;
 
		int minDist = 0x7fffffff;
		int cur = p[0];
 
 
		while (n < p.size())
		{
			int dist;
			while (n < p.size() and (dist = p[n] - cur) < m)
				n++;
 
			if (dist >= m) {
				minDist = min(dist, minDist);
				total++;
			}
 
			if (n >= p.size())
				break;
 
			cur = p[n];
			n++;
		}
 
		if (total < M) {
			e = m - 1;
		} else {
			s = m + 1;
			ret = max(ret, minDist);
		}
	}
 
	return ret;
}
